def enter() -> None:
    print("")
        
num: int = input("enter a positive integer: ")
enter()
        
print(f"{num} =")
enter()
        
small_num: int = 2
result: dict[int, int] = {}
        
while small_num < (num + 1):
    if (num % small_num) == 0:
        result[small_num] = 0
    while (num % small_num) == 0:
        result[small_num] += 1
        num /= small_num
        small_num += 1
        
for prime in result:
    print(f"{prime}^{result[prime]}")
enter()
        
print(result)
