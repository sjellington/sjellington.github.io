def prime_fact(num: int) -> dict:
    small_num: int = 2
    result: dict[int, int] = {}

    while small_num <= num:
        if (num % small_num) == 0:
            result[small_num] = 0
        while (num % small_num) == 0:
            result[small_num] += 1
            num /= small_num
        small_num += 1
    
    return result

def mu(num: int) -> int:
    global golden_fact
    result: int = 0
    fact: dict[int, int] = golden_fact

    if num == 1:
        result = 1
    else:
        for prime in fact:
            if fact[prime] >= 2:
                result = 0
            else:
                r: int = 0
                for prime in fact:
                    r += 1
                result = (-1)**r

    return result

def euclidean(a: int, b: int) -> int:
    consider_num: int = 1
    quotient: int = 1
    residue: int = 1
    result: int = 1

    if a == b:
        result = a
    else:
        if a > b:
            consider_num = a
            quotient = b
        else:
            consider_num = b
            quotient = a
    
    while residue != 0:
        residue = consider_num%quotient
        consider_num = quotient
        quotient = residue
    
    result = consider_num
    return result

def divisors(num: int) -> list[int]:
    result: list[int] = []
    potential: int = 1

    while potential <= num:
        if num % potential == 0:
            result.append(potential)
        potential += 1
    
    return result
